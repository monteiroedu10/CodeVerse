// Defina um vetor de valores para I
I = 0:10:1000;  // Você pode ajustar o intervalo conforme necessário

// Defina o ângulo de fase (em radianos)
theta = 0.8;  // Ângulo em radianos

// Calcule os valores correspondentes de V para cada valor de I usando a equação dada
V = sqrt(2300^2 - (1.1 * I * cos(theta)).^2) - 1.1 * I * sin(theta);

// Plote o gráfico de V em função de I
clf;  // Limpa a figura atual
plot(I, V);
xlabel('Corrente do Estator (A)');
ylabel('Tensão Terminal (V)');
title('Gráfico de V em função de I');
grid on;
